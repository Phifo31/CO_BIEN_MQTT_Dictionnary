59static enum_kv_t* enum_list_from_obj(cJSON *obj){
60 if(!obj || !cJSON_IsObject(obj)) return NULL;
61 enum_kv_t *head = NULL, *tail = NULL;
62 for(cJSON *it = obj->child; it; it = it->next){
63 if(!it->string || !cJSON_IsNumber(it)) continue;
64 enum_kv_t *node = (enum_kv_t*)calloc(1, sizeof(enum_kv_t));
65 if(!node) break;
66 node->key = sdup(it->string);
67 node->value = (int)it->valuedouble;
68 if(!head) head = tail = node; else { tail->next = node; tail = node; }
69 }
70 return head;
71}

99static field_type_t parse_type(const char *s){
100 if(!s) return FT_INT;
101 if(!strcasecmp(s, "int")) return FT_INT;
102 if(!strcasecmp(s, "bool") || !strcasecmp(s, "boolean")) return FT_BOOL;
103 if(!strcasecmp(s, "hex") || !strcasecmp(s, "rgb")) return FT_HEX;
104 if(!strcasecmp(s, "int16")|| !strcasecmp(s,"u16") || !strcasecmp(s,"uint16")) return FT_INT16;
105 if(!strcasecmp(s, "enum") || !strcasecmp(s, "dict")) return FT_ENUM;
106 return FT_INT;
107}

134static bool build_fields_from_node(cJSON *data, field_spec_t **out_arr, size_t *out_n){
135 *out_arr = NULL; *out_n = 0;
136
137 if(cJSON_IsArray(data)){
138 size_t n = cJSON_GetArraySize(data);
139 if(n == 0) return true;
140 field_spec_t *arr = (field_spec_t*)calloc(n, sizeof(field_spec_t));
141 if(!arr) return false;
142
143 size_t k = 0;
144 for(cJSON *it = data->child; it; it = it->next){
145 if(!cJSON_IsObject(it)) continue;
146 cJSON *jname = cJSON_GetObjectItemCaseSensitive(it, "name");
147 cJSON *jtype = cJSON_GetObjectItemCaseSensitive(it, "type");
148 if(!cJSON_IsString(jname) || !cJSON_IsString(jtype)) continue;
149
150 arr[k].name = sdup(jname->valuestring);
151 arr[k].type = parse_type(jtype->valuestring);
152 arr[k].enum_list = NULL;
153
154 if(arr[k].type == FT_ENUM){
155 cJSON *jdict = cJSON_GetObjectItemCaseSensitive(it, "dict");
156 if(!jdict) jdict = cJSON_GetObjectItemCaseSensitive(it, "enum");
157 arr[k].enum_list = enum_list_from_obj(jdict);
158 }
159 k++;
160 }
161 *out_arr = arr; *out_n = k;
162 return true;
163 }
164
165 if(cJSON_IsObject(data)){
166 /* { "field":"int", ... } */
167 size_t n = 0;
168 for(cJSON *it = data->child; it; it = it->next) n++;
169 if(n == 0) return true;
170
171 field_spec_t *arr = (field_spec_t*)calloc(n, sizeof(field_spec_t));
172 if(!arr) return false;
173
174 size_t k = 0;
175 for(cJSON *it = data->child; it; it = it->next){
176 arr[k].name = sdup(it->string);
177 arr[k].type = parse_type(cJSON_IsString(it) ? it->valuestring : "int");
178 arr[k].enum_list = NULL;
179 k++;
180 }
181 *out_arr = arr; *out_n = n;
182 return true;
183 }
184
185 return false;
186}

210bool table_load(table_t *t, const char *json_path){
211 if(!t || !json_path) return false;
212 memset(t, 0, sizeof(*t));
213
214 /* Lire fichier */
215 char *txt = NULL;
216 {
217 FILE *f = fopen(json_path, "rb");
218 if(!f){ LOGE("Ouvrir %s", json_path); return false; }
219 fseek(f, 0, SEEK_END);
220 long L = ftell(f);
221 fseek(f, 0, SEEK_SET);
222 txt = (char*)malloc((size_t)L + 1);
223 if(!txt){ fclose(f); return false; }
224 if(fread(txt, 1, (size_t)L, f) != (size_t)L){ fclose(f); free(txt); return false; }
225 fclose(f);
226 txt[L] = '\0';
227 }
228
229 cJSON *root = cJSON_Parse(txt);
230 free(txt);
231 if(!root){ LOGE("JSON invalide"); return false; }
232
233 /* Collecte des entrées */
234 size_t cap = 16, n = 0;
235 entry_t *arr = (entry_t*)calloc(cap, sizeof(entry_t));
236 if(!arr){ cJSON_Delete(root); return false; }
237
238 /* DFS sur objets/tableaux */
239 cJSON *stack[64]; int sp = 0;
240 stack[sp++] = root;
241
242 while(sp > 0){
243 cJSON *node = stack[--sp];
244
245 if(cJSON_IsObject(node)){
246 cJSON *jtopic = cJSON_GetObjectItemCaseSensitive(node, "topic");
247 cJSON *jdata = cJSON_GetObjectItemCaseSensitive(node, "data");
248 cJSON *jid = cJSON_GetObjectItemCaseSensitive(node, "arbitration_id");
249 if(!jid) jid = cJSON_GetObjectItemCaseSensitive(node, "id");
250
251 if(jtopic && cJSON_IsString(jtopic) && jdata && jid && cJSON_IsNumber(jid)){
252 if(n == cap){
253 size_t new_cap = cap * 2;
254 void *tmp = realloc(arr, new_cap * sizeof(entry_t));
255 if(!tmp){ free(arr); cJSON_Delete(root); return false; }
256 arr = (entry_t*)tmp;
257 memset(&arr[cap], 0, (new_cap - cap) * sizeof(entry_t));
258 cap = new_cap;
259 }
260
261 entry_t *e = &arr[n];
262 memset(e, 0, sizeof(*e));
263 e->topic = sdup(jtopic->valuestring);
264 e->can_id = (uint32_t)jid->valuedouble;
265
266 if(!build_fields_from_node(jdata, &e->fields, &e->field_count)){
267 LOGW("data invalide pour %s", e->topic ? e->topic : "(null)");
268 } else {
269 n++;
270 }
271 }
272
273 for(cJSON *it = node->child; it; it = it->next){
274 if(cJSON_IsObject(it) || cJSON_IsArray(it)) stack[sp++] = it;
275 }
276 } else if(cJSON_IsArray(node)){
277 for(cJSON *it = node->child; it; it = it->next){
278 if(cJSON_IsObject(it) || cJSON_IsArray(it)) stack[sp++] = it;
279 }
280 }
281 }
282
283 cJSON_Delete(root);
284
285 t->entries = arr;
286 t->entry_count = n;
287 LOGI("Table chargée: %zu topics, %zu IDs", n, n);
288
289 return (n > 0);
290}

298void table_free(table_t *t){
299 if(!t || !t->entries) return;
300 for(size_t i = 0; i < t->entry_count; i++){
301 entry_t *e = &t->entries[i];
302 free(e->topic);
303 if(e->fields){
304 for(size_t k = 0; k < e->field_count; k++){
305 free(e->fields[k].name);
306 enum_list_free(e->fields[k].enum_list);
307 }
308 free(e->fields);
309 }
310 }
311 free(t->entries);
312 memset(t, 0, sizeof(*t));
313}
