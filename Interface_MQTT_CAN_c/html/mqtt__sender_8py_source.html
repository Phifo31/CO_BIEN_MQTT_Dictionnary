67 def walk(node):
68 if isinstance(node, dict):
69 # bloc avec topic + data => une "entrée"
70 if "topic" in node and "data" in node and isinstance(node["data"], dict):
71 topic = str(node["topic"]).rstrip("/")
72 data = node["data"]
73 # normaliser la description des champs: {"name":..., "type":..., "dict":...}
74 fields = []
75 for k, v in data.items():

78 elif isinstance(v, dict) and v: # enum/dict (ex: mode)
79 fields.append({"name": k, "type": "enum", "dict": v})
80 else:
81 # champ vide: on l’ignore (payload n’en aura pas)
82 pass

84 # continuer à explorer
85 for v in node.values():
86 if isinstance(v, (dict, list)):
87 walk(v)
88 elif isinstance(node, list):

95def build_payload(entry):
96 payload = {}
97 for f in entry["fields"]:
98 name, typ, edict = f["name"], f["type"], f.get("dict")
99 val = rand_for_field(name, typ, edict)
100 payload[name] = val
101 # petites règles de cohérence (ex: intensity bornée)
102 if "intensity" in payload:
103 try: payload["intensity"] = clamp(int(payload["intensity"]), 0, 255)
104 except: payload["intensity"] = 128
