110bool pack_payload(uint8_t out8[8], const entry_t *entry, cJSON *json_in){
111 memset(out8,0,8);
112 if(!entry || !json_in) return false;
113 size_t idx=0;
114
115 for(size_t i=0;i<entry->field_count;i++){
116 const field_spec_t *fs = &entry->fields[i];
117 cJSON *v = cJSON_GetObjectItemCaseSensitive(json_in, fs->name);
118 if(!v){
119 LOGW("Champ manquant: %s", fs->name);
120 return false;
121 }
122 switch(fs->type){
123 case FT_INT:{
124 if(!cJSON_IsNumber(v)) { LOGW("Type int attendu pour %s", fs->name); return false; }
125 if(idx+1>8) return false;
126 int x = (int)v->valuedouble;
127 if(x<0 || x>255){ LOGW("Valeur %s hors plage: %d", fs->name, x); return false; }
128 out8[idx++] = (uint8_t)x;
129 }break;
130 case FT_BOOL:{
131 if(!cJSON_IsBool(v)) { LOGW("Type bool attendu pour %s", fs->name); return false; }
132 if(idx+1>8) return false;
133 out8[idx++] = cJSON_IsTrue(v) ? 1 : 0;
134 }break;
135 case FT_HEX:{
136 if(!cJSON_IsString(v)) { LOGW("Type hex(#RRGGBB) attendu pour %s", fs->name); return false; }
137 if(idx+3>8) return false;
138 uint8_t rgb[3];
139 if(!parse_hex_rgb(v->valuestring, rgb)){ LOGW("Format hex invalide pour %s", fs->name); return false; }
140 out8[idx++]=rgb[0]; out8[idx++]=rgb[1]; out8[idx++]=rgb[2];
141 }break;
142 case FT_INT16:{
143 if(!cJSON_IsNumber(v)){ LOGW("Type int16 attendu pour %s", fs->name); return false; }
144 if(idx+2>8) return false;
145 int x = (int)v->valuedouble;
146 if(x<0 || x>65535){ LOGW("Valeur %s hors plage: %d", fs->name, x); return false; }
147 out8[idx++] = (uint8_t)((x>>8)&0xFF);
148 out8[idx++] = (uint8_t)(x & 0xFF);
149 }break;
150 case FT_ENUM:{
151 if(!cJSON_IsString(v)){ LOGW("Type enum(string) attendu pour %s", fs->name); return false; }
152 if(idx+1>8) return false;
153 uint8_t code=0;
154 if(!enum_str_to_code(fs, v->valuestring, &code)){
155 LOGW("Valeur enum inconnue '%s' pour %s", v->valuestring, fs->name);
156 return false;
157 }
158 out8[idx++] = code;
159 }break;
160 }
161 }
162 /* Les octets restants sont à 0 par défaut */
163 return true;
164}

184cJSON* unpack_payload(const uint8_t in8[8], const entry_t *entry){
185 if(!entry) return NULL;
186 cJSON *obj = cJSON_CreateObject();
187 if(!obj) return NULL;
188 size_t idx=0;
189
190 for(size_t i=0;i<entry->field_count;i++){
191 const field_spec_t *fs = &entry->fields[i];
192 switch(fs->type){
193 case FT_INT:{
194 if(idx+1>8) goto fail;
195 cJSON_AddNumberToObject(obj, fs->name, (int)in8[idx++]);
196 }break;
197 case FT_BOOL:{
198 if(idx+1>8) goto fail;
199 cJSON_AddBoolToObject(obj, fs->name, in8[idx++] ? 1:0);
200 }break;
201 case FT_HEX:{
202 if(idx+3>8) goto fail;
203 char buf[8]; snprintf(buf,sizeof(buf),"#%02X%02X%02X", in8[idx],in8[idx+1],in8[idx+2]);
204 cJSON_AddStringToObject(obj, fs->name, buf); idx+=3;
205 }break;
206 case FT_INT16:{
207 if(idx+2>8) goto fail;
208 int v = ((int)in8[idx]<<8) | (int)in8[idx+1]; idx+=2;
209 cJSON_AddNumberToObject(obj, fs->name, v);
210 }break;
211 case FT_ENUM:{
212 if(idx+1>8) goto fail;
213 uint8_t code = in8[idx++];
214 const char *s = enum_code_to_str(fs, code);
215 if(s) cJSON_AddStringToObject(obj, fs->name, s);
216 else cJSON_AddNumberToObject(obj, fs->name, (int)code);
217 }break;
218 }
219 }
220 return obj;
221fail:
222 cJSON_Delete(obj);
223 return NULL;
224}
