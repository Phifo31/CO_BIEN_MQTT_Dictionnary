31def load_entries(path):
32 obj = json.loads(Path(path).read_text(encoding="utf-8"))
33 out = []
34 def walk(node):
35 if isinstance(node, dict):
36 if "topic" in node and isinstance(node.get("data"), list) and "id" in node:
37 out.append({"topic": node["topic"].rstrip("/"), "data": node["data"], "id": int(node["id"])})
38 for v in node.values():
39 if isinstance(v, (dict, list)): walk(v)
40 elif isinstance(node, list):
41 for it in node: walk(it)
42 walk(obj)

74def pack8(entry, payload):
75 out = [0]*8; idx = 0
76 for fs in entry["data"]:
77 name = (fs.get("name") or fs.get("field"))
78 t = (fs.get("type") or "").lower()
79 v = payload.get(name, None)
80 if t in ("int","uint8","byte"):
81 if v is None: return None
82 v = int(v);
83 if v<0 or v>255: return None
84 if idx+1>8: return None
85 out[idx]=v; idx+=1
86 elif t in ("bool","boolean"):
87 if v is None: return None
88 out[idx]= 1 if bool(v) else 0; idx+=1
89 elif t in ("hex","rgb"):
90 rgb = parse_hex_rgb(v)
91 if not rgb or idx+3>8: return None
92 out[idx:idx+3] = rgb; idx+=3
93 elif t in ("int16","i16","uint16","u16"):
94 if v is None: return None
95 n = int(v) & 0xFFFF
96 if idx+2>8: return None
97 out[idx] = (n>>8)&0xFF; out[idx+1] = n&0xFF; idx+=2
98 elif t in ("enum","dict"):
99 if not isinstance(v,str): return None
100 code = enum_str_to_code(fs, v)
101 if code is None or idx+1>8: return None
102 out[idx]=code & 0xFF; idx+=1
103 else:
104 return None
105 return bytes(out)
