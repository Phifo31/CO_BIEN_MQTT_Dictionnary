93bool my_setup(const char *cfg_path) {
94 signal(SIGINT, on_sig);
95 signal(SIGTERM, on_sig);
96
97 /* Réinitialisation mémoire des structures globales */
98 memset(&g_table, 0, sizeof(g_table));
99 memset(&g_mqtt, 0, sizeof(g_mqtt));
100 memset(&g_can, 0, sizeof(g_can));
101
102 /* Chargement du fichier de conversion */
103 if (!table_load(&g_table, cfg_path)) {
104 LOGE("Echec chargement table: %s", cfg_path);
105 return false;
106 }
107
108 /* Initialisation du client MQTT */
109 if (!mqtt_init(&g_mqtt, MQTT_HOST, MQTT_PORT, 60))
110 return false;
111
112 mqtt_set_qos(&g_mqtt, 1, 1);
113
114 /* Liaison des modules entre eux (Lier la callback MQTT -> CAN avec userdata (table+can+mqtt) */
115 struct { const table_t *t; can_ctx_t *c; mqtt_ctx_t *m; } *ub =
116 malloc(sizeof(*ub));
117 if (!ub) return false;
118
119 ub->t = &g_table; ub->c = &g_can; ub->m = &g_mqtt;
120 mqtt_set_user_data(&g_mqtt, ub);
121
122 /* Abonnement à tous les topics MQTT (sans doublon local) */
123 if (!mqtt_subscribe_all_nolocal(&g_mqtt)) return false;
124
125 /* Initialisation du bus CAN */
126 if (!can_init(&g_can, IFNAME)) return false;
127
128 LOGI("Setup OK (cfg=%s, if=%s, mqtt=%s:%d)", cfg_path, IFNAME, MQTT_HOST, MQTT_PORT);
129 return true;
130}
