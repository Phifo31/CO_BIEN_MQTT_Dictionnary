56/* MQTT -> CAN (mode “tunnel”) */
57static void on_message(struct mosquitto *m, void *ud, const struct mosquitto_message *msg){
58 (void)m;
59 if(!ud || !msg || !msg->topic) return;
60
61 user_bundle_t *ub = (user_bundle_t*)ud;
62 if(!ub->table || !ub->can) return;
63
64 char base[256]; topic_base_from_input(msg->topic, base, sizeof(base));
65 if(base[0]=='\0') return; /* /state ignoré */
66
67 const entry_t *e = table_find_by_topic(ub->table, base);
68 if(!e){
69 LOGW("Topic inconnu: %s", msg->topic);
70 return;
71 }
72
73 /* Parse JSON */
74 cJSON *in = NULL;
75 if(msg->payload && msg->payloadlen>0){
76 char *buf = (char*)malloc((size_t)msg->payloadlen+1);
77 if(!buf) return;
78 memcpy(buf, msg->payload, (size_t)msg->payloadlen);
79 buf[msg->payloadlen]='\0';
80 in = cJSON_Parse(buf);
81 free(buf);
82 }
83 if(!in){
84 LOGW("Payload JSON invalide sur %s", msg->topic);
85 return;
86 }
87
88 /* 1) On packe le “corps” jusqu’à 8 octets */
89 uint8_t body[8]={0};
90 bool ok_body = pack_payload(body, e, in);
91 cJSON_Delete(in);
92 if(!ok_body){
93 LOGE("Pack échoué pour topic %s", base);
94 return;
95 }
96
97 /* 2) Mode tunnel : [MSB(inner_id), LSB(inner_id), body[0..5]] */
98 uint8_t out8[8]={0};
99 out8[0] = (uint8_t)((e->can_id >> 8) & 0xFF);
100 out8[1] = (uint8_t)( e->can_id & 0xFF);
101 memcpy(out8+2, body, 6); /* on place au plus 6 octets derrière */
102
103 /* 3) Envoi sur l’ID de transport attendu par la STM32 */
104 if(!can_send(ub->can, BRIDGE_TUNNEL_CANID, out8)){
105 LOGE("Envoi CAN échoué (transport=0x%X, inner_id=0x%X)",
106 BRIDGE_TUNNEL_CANID, e->can_id);
107 return;
108 }
109 LOGI("MQTT->CAN OK topic=%s transport=0x%X inner_id=0x%X",
110 base, BRIDGE_TUNNEL_CANID, e->can_id);
111}

137 LOGE("mosquitto_connect");
