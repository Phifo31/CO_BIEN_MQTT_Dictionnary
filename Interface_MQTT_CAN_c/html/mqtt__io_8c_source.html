96static void topic_base_from_input(const char *in, char *base, size_t n){
97 if(!in||!base||!n) return;
98 if(ends_with(in,"/state")){ base[0]='\0'; return; } /* ignoré */
99 if(ends_with(in,"/cmd")){
100 size_t L=strlen(in);
101 size_t nb=(L>=4)?(L-4):0;
102 if(nb>=n) nb=n-1;
103 memcpy(base,in,nb); base[nb]='\0'; return;
104 }
105 snprintf(base,n,"%s",in);
106}

155static void on_message(struct mosquitto *m, void *ud, const struct mosquitto_message *msg){
156 (void)m;
157 if(!ud || !msg || !msg->topic) return;
158
159 user_bundle_t *ub = (user_bundle_t*)ud;
160 if(!ub->table || !ub->can) return;
161
162 char base[256]; topic_base_from_input(msg->topic, base, sizeof(base));
163 if(base[0]=='\0') return; /* /state ignoré */
164
165 const entry_t *e = table_find_by_topic(ub->table, base);
166 if(!e){
167 LOGW("Topic inconnu: %s", msg->topic);
168 return;
169 }
170
171 /* Lecture du JSON reçu */
172 cJSON *in = NULL;
173 if(msg->payload && msg->payloadlen>0){
174 char *buf = (char*)malloc((size_t)msg->payloadlen+1);
175 if(!buf) return;
176 memcpy(buf, msg->payload, (size_t)msg->payloadlen);
177 buf[msg->payloadlen]='\0';
178 in = cJSON_Parse(buf);
179 free(buf);
180 }
181 if(!in){
182 LOGW("Payload JSON invalide sur %s", msg->topic);
183 return;
184 }
185
186 /* Conversion JSON → binaire */
187 uint8_t body[8]={0};
188 bool ok_body = pack_payload(body, e, in);
189 cJSON_Delete(in);
190 if(!ok_body){
191 LOGE("Pack échoué pour topic %s", base);
192 return;
193 }
194
195 /* Construction du message tunnel : [ID haut, ID bas, data...] */
196 uint8_t out8[8]={0};
197 out8[0] = (uint8_t)((e->can_id >> 8) & 0xFF);
198 out8[1] = (uint8_t)( e->can_id & 0xFF);
199 memcpy(out8+2, body, 6); /* on place au plus 6 octets derrière */
200
201 /* Envoi sur le bus CAN */
202 if(!can_send(ub->can, BRIDGE_TUNNEL_CANID, out8)){
203 LOGE("Envoi CAN échoué (transport=0x%X, inner_id=0x%X)",
204 BRIDGE_TUNNEL_CANID, e->can_id);
205 return;
206 }
207 LOGI("MQTT->CAN OK topic=%s transport=0x%X inner_id=0x%X",
208 base, BRIDGE_TUNNEL_CANID, e->can_id);
209}

227bool mqtt_init(mqtt_ctx_t *ctx, const char *host, int port, int keepalive){
228 if(!ctx) return false;
229 memset(ctx,0,sizeof(*ctx));
230 ctx->qos_pub=1; ctx->qos_sub=1;
231
232 mosquitto_lib_init();
233 ctx->mosq = mosquitto_new(NULL, true, NULL);
234 if(!ctx->mosq){
235 LOGE("mosquitto_new");
236 return false;
237 }
238
239
240/* Utilisation de MQTT v5 pour utiliser l’option “no_local” (anti-doublon) */
241 mosquitto_int_option(ctx->mosq, MOSQ_OPT_PROTOCOL_VERSION, MQTT_PROTOCOL_V5);
242
243 mosquitto_connect_callback_set(ctx->mosq, on_connect);
244 mosquitto_disconnect_callback_set(ctx->mosq, on_disconnect);
245 mosquitto_message_callback_set(ctx->mosq, on_message);
246
247 if(mosquitto_connect(ctx->mosq,
248 host?host:"localhost",
249 port>0?port:1883,
250 keepalive>0?keepalive:60) != MOSQ_ERR_SUCCESS){
251 LOGE("mosquitto_connect");
252 mosquitto_destroy(ctx->mosq);
253 ctx->mosq=NULL;
254 mosquitto_lib_cleanup();
255 return false;
256 }
257 return true;
258}

366 LOGE("cJSON_PrintUnformatted");

368 }
369
370 bool ok = mqtt_publish_json(ctx, e->topic, out); /* publier sur le topic de base */
371 free(out);
372 if(ok) LOGI("CAN->MQTT OK id=0x%X topic=%s", e->can_id, e->topic);
373 else LOGE("CAN->MQTT publish échoué topic=%s", e->topic);
374 return ok;
375}
