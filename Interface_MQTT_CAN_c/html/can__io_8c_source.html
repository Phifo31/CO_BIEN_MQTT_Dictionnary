69bool can_init(can_ctx_t *c, const char *ifname){
70 if (!c || !ifname) return false;
71 memset(c, 0, sizeof(*c));
72 c->fd = -1;
73
74 /* Création de la socket CAN brute */
75 int fd = socket(PF_CAN, SOCK_RAW, CAN_RAW);
76 if (fd < 0){ LOGE("socket(PF_CAN): %s", strerror(errno)); return false; }
77
78 /* Recherche de l’index de l’interface */
79 struct ifreq ifr; memset(&ifr, 0, sizeof(ifr));
80 snprintf(ifr.ifr_name, sizeof(ifr.ifr_name), "%s", ifname);
81 if (ioctl(fd, SIOCGIFINDEX, &ifr) < 0){
82 LOGE("SIOCGIFINDEX(%s): %s", ifname, strerror(errno));
83 close(fd); return false;
84 }
85
86 /* Ne pas recevoir nos propres trames (supprime les doublons) */
87 int recv_own = 0; // 0 = off
88 if (setsockopt(fd, SOL_CAN_RAW, CAN_RAW_RECV_OWN_MSGS, &recv_own, sizeof(recv_own)) < 0) {
89 LOGW("setsockopt(CAN_RAW_RECV_OWN_MSGS): %s", strerror(errno));
90 }
91
92 /* Liaison de la socket à l’interface CAN */
93 struct sockaddr_can addr; memset(&addr, 0, sizeof(addr));
94 addr.can_family = AF_CAN;
95 addr.can_ifindex = ifr.ifr_ifindex;
96
97 if (bind(fd, (struct sockaddr*)&addr, sizeof(addr)) < 0){
98 LOGE("bind CAN(%s): %s", ifname, strerror(errno));
99 close(fd); return false;
100 }
101
102 /* Mode non bloquant */
103 if (!set_nonblock(fd)){
104 LOGW("fcntl(O_NONBLOCK) échoué (socket bloquant)");
105 }
106
107 /* Agrandir les buffers de réception/émission */
108 int rcvbuf = 256*1024;
109 (void)setsockopt(fd, SOL_SOCKET, SO_RCVBUF, &rcvbuf, sizeof(rcvbuf));
110 int sndbuf = 256*1024;
111 (void)setsockopt(fd, SOL_SOCKET, SO_SNDBUF, &sndbuf, sizeof(sndbuf));
112
113 c->fd = fd;
114 return true;
115}

131bool can_send(can_ctx_t *c, uint32_t can_id, const uint8_t data[8]){
132 if (!c || c->fd < 0) return false;
133
134 struct can_frame f; memset(&f, 0, sizeof(f));
135 f.can_id = can_id & CAN_SFF_MASK;
136 f.can_dlc = 8;
137
138 memcpy(f.data, data, 8);
139
140 ssize_t n = write(c->fd, &f, sizeof(f));
141 if (n < 0){
142 LOGE("CAN write: %s", strerror(errno));
143 return false;
144 }
145 return (n == (ssize_t)sizeof(f));
146}

170void can_poll(can_ctx_t *c, const table_t *t, mqtt_ctx_t *m, int max_frames){
171 if (!c || c->fd < 0) return;
172 if (max_frames <= 0) max_frames = 8;
173
174 fd_set rfds; struct timeval tv = (struct timeval){0,0};
175 FD_ZERO(&rfds); FD_SET(c->fd, &rfds);
176
177 /* Vérifie s’il y a des données à lire sans bloquer */
178 int rv = select(c->fd + 1, &rfds, NULL, NULL, &tv);
179 if (rv <= 0 || !FD_ISSET(c->fd, &rfds)) return;
180
181 for (int i = 0; i < max_frames; i++){
182 struct can_frame f; ssize_t n = read(c->fd, &f, sizeof(f));
183 if (n < 0){
184 if (errno == EAGAIN || errno == EWOULDBLOCK) break;
185 LOGW("CAN read: %s", strerror(errno));
186 break;
187 }
188 if ((size_t)n != sizeof(f)) break;
189
190 /* 1) tentative d'identification par ID CAN */
191 const entry_t *e = t ? table_find_by_canid(t, f.can_id) : NULL;
192 const uint8_t *payload = f.data;
193 uint8_t shifted[8];
194
195 /* 2) Si echec, on tente le mode tunnel (ID dans les 2 premiers octets) */
196 if (!e && f.can_dlc >= 2){
197 uint16_t inner_id = ((uint16_t)f.data[0] << 8) | f.data[1];
198 e = table_find_by_canid(t, inner_id);
199 if (e){
200 /* on “retire” les 2 octets d’en-tête et on recale le payload */
201 memset(shifted, 0, sizeof(shifted));
202 size_t copy = (f.can_dlc > 2) ? (size_t)(f.can_dlc - 2) : 0;
203 if (copy > 6) copy = 6; /* place restante (8-2) */
204 memcpy(shifted, f.data + 2, copy);
205 payload = shifted;
206 }
207 }
208 /* 3) Si l’entrée correspond, on renvoie vers MQTT */
209 if (!e) continue;
210 (void)mqtt_handle_can_message(m, e, payload);
211 }
212}
